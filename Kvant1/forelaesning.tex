	\begin{equation}
		[\op{A},\op{B}] = \op{A}\op{B} - \op{B} \op{A}
	\end{equation}
	kanonisk kommutator
	\begin{equation}
		[\op{x},\op{p}] = i \hbar
	\end{equation}
	
	Nu:
	\begin{itemize}
		\item $ \haev $ og $ \saenk $ er \textit{ikke} hinandens kompleks konjugerede. Men de er hinandens hermitisk konjugerede (adjungerede fra LinAlg). Eksempel. få $ d_n $ fra $ \saenk\psi_n = d_n \psi_n $ (står også i bogen)
		\begin{align}
			\int (\saenk \psi_n)\konj (\saenk \psi_n) \ud x &= \int (d_n \psi_n)\konj (d_n \psi_n) \ud x = |d_n|^2 \\
			\int \psi_n \konj \haev \saenk \psi_n \ud x &= n \int |\psi_n|^2\ud x = n, \\
			|d_n|^2 = n, \quad \Rightarrow \quad d_n &= \sqrt{n}, \ \text{vælger positiv rod}
		\end{align}
