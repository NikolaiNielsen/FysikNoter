\section*{Regneregler for integration}
 \begin{align*}
 \int (f \pm g)\,\mathrm{d}x &=\int f \,\mathrm{d}x \pm \int g \,\mathrm{d}x\\
 \int k\cdot f \,\mathrm{d}x &= k\int f \,\mathrm{d}\\
 \int_a^b f \,\mathrm{d} &= \left[ F \right]_a^b = F(b)-F(a) 
\end{align*}
\section*{Partiel integration}
\begin{align*}
\int f(x)\cdot g(x)\,\mathrm{d}x=F(x)\cdot g(x) - \int F(x)\cdot g'(x)\,\mathrm{d}x
\end{align*}
\section*{integration ved substitution}
\begin{align*}
\int_a^b f(g(x))\,\mathrm{d}x
\end{align*}
definerer $g(x)=u$, udregner de nye grænser\footnote{kun nødvendigt ved bestemte integraler} $g(a)$ og $g(b)$, og finder $\frac{\mathrm{d}u}{\mathrm{d}x}$:
\begin{align*}
\frac{\mathrm{d}u}{\mathrm{d}x}=g'(x) \Rightarrow \,\mathrm{d}x=\frac{\mathrm{d}u}{g'(x)}
\end{align*}
Det hele indsættes nu, og følgende integral udregnes:
\begin{align*}
\int_{g(a)}^{g(b)} f(u) \,\mathrm{d}u
\end{align*}
