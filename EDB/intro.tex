	\subsection*{Vektorafledte}
	\paragraph{Kartesiske koordinater} \quad $ \ud \V{l} = \uuud x \,\Vx + \uuud y \, \Vy + \uuud z \,\Vz  $, \quad $ \ud \tau =  \ud x \ud y \ud z $.
	
	\vspace{0.3cm}
	\begin{tabular}[H]{>{\itshape}l >{$\displaystyle} l <{$} c  >{$\displaystyle} l <{$} }
		\vspace{0.3cm}
		Gradient \textup{:}		& \grad t 				& = & \diff{t}{x} \Vx+ \diff{t}{y} \Vy + \diff{t}{z} \Vz \\
		\vspace{0.3cm}
		Divergens \textup{:} 	& \grad \D \V{v} 		& = & \diff{v_x}{x} + \diff{v_y}{y} + \diff{v_z}{z} \\
		\vspace{0.3cm}
		Rotation \textup{:}		& \grad \times \V{v}	& = & \begin{vmatrix}
			\Vx        & \Vy        & \Vz        \\
			\partial x & \partial y & \partial z \\
			v_x        & v_y        & v_z
		\end{vmatrix} = \pp{ \diff{v_z}{y} - \diff{v_y}{z} } \Vx + \pp{ \diff{v_x}{z} - \diff{v_z}{x} } \Vy + \pp{ \diff{v_y}{x} - \diff{v_x}{y} } \Vz \\
		Laplace \textup{:} 		& \grad^2 t 			& = & \diff{^2t}{x^2} + \diff{^2t}{y^2} + \diff{^2t}{z^2}
	\end{tabular}
	
	\paragraph{Sfæriske koordinater} \quad $ \ud \V{l} = \uuud r \,\Vr + r\uuud \theta \, \Vt +r \sin \theta \uuud \phi \,\Vp  $, \quad $ \ud \tau =  r^2 \sin \theta\ud r \ud \theta \ud \phi $.
	
	\vspace{0.3cm}
	\begin{tabular}[H]{>{\itshape}l >{$\displaystyle} l <{$} c  >{$\displaystyle} l <{$} }
		\vspace{0.3cm}
		Gradient \textup{:}		& \grad t 				& = & \diff{t}{r} \Vr+ \frac{1}{r}\diff{t}{\theta} \Vt + \frac{1}{r \sin \theta}\diff{t}{\phi} \Vp \\
		\vspace{0.3cm}
		Divergens \textup{:} 	& \grad \D \V{v} 		& = & \frac{1}{r^2}\diff{}{r}(r^2 v_r) + \frac{1}{r \sin \theta} \diff{}{\theta} (\sin \theta \ v_{\theta}) + \frac{1}{r \sin \theta}\diff{v_{\phi}}{\phi} \\
		\vspace{0.3cm}
		Rotation \textup{:}		& \grad \times \V{v}	& = &  \frac{1}{r \sin \theta} \bb{ \diff{}{\theta}(\sin \theta \ v_{\phi}) - \diff{v_{\theta}}{\phi} } \Vr + \frac{1}{r}\bb{ \frac{1}{\sin \theta}\diff{v_r}{\phi} - \diff{}{r}(r \ v_{\phi}) } \Vt \\
		& & & + \frac{1}{r}\bb{ \diff{}{r} (r \ v_{\theta}) - \diff{v_r}{\theta} } \Vp \vspace{0.3cm}\\
		Laplace \textup{:} 		& \grad^2 t 			& = & \frac{1}{r^2}\diff{}{r} \pp{r^2 \diff{t}{r}} + \frac{1}{r^2 \sin \theta} \diff{}{\theta}\pp{\sin \theta \diff{t}{\theta}}+  \frac{1}{r^2 \sin^2 \theta}\diff{^2t}{\phi^2}
	\end{tabular}
	
	\paragraph{Cylinder koordinater} \quad $ \ud \V{l} = \uuud s \,\U{s} + s\uuud \phi \, \Vp + \uuud z \,\Vz  $, \quad $ \ud \tau =  s \ud s \ud \phi \ud z $.

	\vspace{0.3cm}
	\begin{tabular}[H]{>{\itshape}l >{$\displaystyle} l <{$} c  >{$\displaystyle} l <{$} }
		\vspace{0.3cm}
		Gradient \textup{:}		& \grad t 				& = & \diff{t}{s} \U{s}+ \frac{1}{s} \diff{t}{\phi} \Vp + \diff{t}{z} \Vz \\
		\vspace{0.3cm}
		Divergens \textup{:} 	& \grad \D \V{v} 		& = & \frac{1}{s}\diff{}{s}(s v_s) +  \frac{1}{s} \diff{v_{\phi}}{\phi} + \diff{v_z}{z} \\
		\vspace{0.3cm}
		Rotation \textup{:}		& \grad \times \V{v}	& = & \bb{ \frac{1}{s}\diff{v_z}{\phi} - \diff{v_{\phi}}{z} } \U{s} + \bb{ \diff{v_s}{z} - \diff{v_z}{s} } \Vp + \frac{1}{s}\bb{ \diff{}{s}(s \ v_{\phi}) - \diff{v_s}{\phi} } \Vz \\
		Laplace \textup{:} 		& \grad^2 t 			& = & \frac{1}{s} \diff{}{s} \pp{s \diff{t}{s}} + \frac{1}{s^2}\diff{^2t}{\phi^2} + \diff{^2t}{z^2}
	\end{tabular}
	
	\subsection*{Vektoridentiteter}
	\paragraph{Tripleprodukter}
	\begin{enumerate}
		\item $ \V{A} \D (\V{B} \times \V{C}) =  \V{B} \D (\V{C \times \V{A}}) = \V{C} \D (\V{A} \times \V{B}) $ 
		\item $ \V{A} \times (\V{B} \times \V{C}) = \V{B}(\V{A} \D \V{C}) -\V{C}( \V{A} \D \V{B} ) $
		\setcounter{VektorListe}{\theenumi}
	\end{enumerate}
	\paragraph{Produktregler}
	\begin{enumerate}
		\setcounter{enumi}{\theVektorListe}
		\item $ \grad (fg) = f(\grad g) + g(\grad f) $
		\item $ \grad (\V{A} \D \V{B}) = \V{A} \times (\grad \times \V{B}) +\V{B} \times (\grad \times \V{A}) + (\V{A} \D \grad)\V{B} + (\V{B} \D \grad)\V{A} $
		\item $ \grad \D (f\V{A}) = f(\grad \D \V{A}) + \V{A} \D (\grad f) $
		\item $ \grad \D (\V{A} \times \V{B}) = \V{B} \D (\grad \times \V{A}) - \V{A} \D (\grad \times \V{B}) $
		\item $ \grad \times (\V{A} \times \V{B}) = (\V{B} \D \grad) \V{A} - (\V{A} \D \grad)\V{B} + \V{A}(\grad \D \V{B}) - \V{B} (\grad \D \V{A}) $
		\setcounter{VektorListe}{\theenumi}
	\end{enumerate}
	\paragraph{Andenafledte}
	\begin{enumerate}
		\setcounter{enumi}{\theVektorListe}
		\item $ \grad \D (\grad \times \V{A}) = 0 $
		\item $ \grad \times (\grad f) = 0 $
		\item $ \grad \times (\grad \times \V{A}) = \grad(\grad \D \V{A}) - \grad^2 \V{A} $
	\end{enumerate}
